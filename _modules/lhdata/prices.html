def _sql_px_orig_all(tb_template: str,
                     fields: Union[str, list, tuple] = '*',
                     day: str = None, start: str = None, end: str = None) -> list:
    """
    SQL, 查询所有股票的原始价格。

    :param tb_template: 表名模板，格式类似为'hisquotation_orig_sz.tb_hisbar_day_{year}'
    :param fields:
    :param day:
    :param start:
    :param end:
    :return: sql list
    """
    if day:
        start = end = day

    sql_template = f"""SELECT {__px_fields_str(fields=fields)} FROM {tb_template}
                            WHERE close > 0
                            {_date_str(day=day, start=start, end=end, key='date1')}
                            ORDER BY date1 ASC"""
    start = start or DEFAULT_START
    end = end or pd.Timestamp.now().date().isoformat()
    year_list = list(range(int(start.split('-')[0]) // 5 * 5,
                           int(end.split('-')[0]) // 5 * 5 + 1,
                           5))
    sql_list = [sql_template.format(year=y) for y in year_list]
    return sql_list


def _sql_px_orig_single(symbol: str, tb_template: str,
                        fields: Union[str, list, tuple] = '*',
                        day: str = None, start: str = None, end: str = None) -> list:
    """
    SQL, 查询单支股票的原始价格。

    :param symbol： 格式为'000001 CH Equity' 或 'SHSZ300 Index'
    :param tb_template: 表名模板，格式类似为'hisquotation_orig_sz.tb_hisbar_day_{year}'
    :param fields:
    :param day:
    :param start:
    :param end:
    :return: sql list
    """
    if day:
        start = end = day

    sql_template = f"""SELECT {__px_fields_str(fields=fields)} FROM {tb_template}
                            WHERE symbol = '{symbol}'
                            AND close > 0
                            {_date_str(day=day, start=start, end=end, key='date1')}
                            ORDER BY date1 ASC"""
    start = start or DEFAULT_START
    end = end or pd.Timestamp.now().date().isoformat()
    year_list = list(range(int(start.split('-')[0]) // 5 * 5,
                           int(end.split('-')[0]) // 5 * 5 + 1,
                           5))
    sql_list = [sql_template.format(year=y) for y in year_list]
    return sql_list


def _sql_px_orig_multiple(symbol_list: Union[list, tuple], tb_template: str,
                          fields: Union[str, list, tuple] = '*',
                          day: str = None, start: str = None, end: str = None) -> list:
    """
    SQL, 查询多支股票的原始价格。

    :param symbol_list： 格式为'000001 CH Equity' 或 'SHSZ300 Index'
    :param tb_template: 表名模板，格式类似为'hisquotation_orig_sz.tb_hisbar_day_{year}'
    :param fields:
    :param day:
    :param start:
    :param end:
    :return: sql list
    """
    return [_sql_px_orig_single(symbol=s, tb_template=tb_template,
                                fields=fields, day=day, start=start, end=end)
            for s in symbol_list]

[docs]def get_px_orig(symbol: Union[str, list, tuple], day: str = None, start: str = None, end: str = None,
                fields: Union[str, list, tuple] = '*', incl_prev: bool = False,
                db_config: dict = None, db_name: str = None, tb_name: str = None,
                hk: bool = False, area: str = None, progress: bool = False,
                nbr_thread: int = 0, chunksize: int = 1,
                **kwargs):
    """
    获取原始行情。建议使用get_px接口。

    可指定db读取行情，该db行情表需与161行情表一致，表名相同，且每五年分一个表。

    :param symbol: 如'000001 CH Equity'；空('' 或 [] 或 () ) 代表所有股票；支持股票列表或股票元组。
    :param start: 起始日期（含），格式如'2019-01-01'；空('' 或 None) 则自动设置为1992-01-01。
    :param end: 终止日期（含），格式如'2019-09-01'；空('' 或 None) 则自动设置为查询当日。
    :param day: 指定日期，等效于设置了start=end=day, 会忽略start、end的输入。
    :param fields: 指定字段，如'close' 或 ['open', 'close']; 会自动添加‘datetime’和‘symbol’两个字段。默认为所有字段。
    :param incl_prev: 包含start前一个正常交易日的数据，针对停牌日需要用价格填充的情况。只支持单个股票或股票列表。
    :param db_config: 指定数据库连接，None则自动分配。
    :param db_name: 指定数据库名，None则自动分配。
    :param tb_name: 指定的表名，格式为{db_name}.{tb_name}，None则自动分配。
    :param hk: 在香港使用需设置为True。
    :param area: in ['CH', 'US', ...]。指定symbol区域，不指定则根据symbol（或symbol list的第一个）判定。为空时，自动设置为'CH'。
    :param progress: 显示下载进度条。
    :param nbr_thread: 使用多少个进程。0代表自动；1或负值代表不使用多进程。
    :param chunksize: 每个进程处理多少个查询。默认为1。
    :param kwargs:

    :return: a list/tuple of symbols -> DataFrame list; otherwise, DataFrame
    """
    if incl_prev and (not symbol):
        raise NotImplementedError(f"当incl_prev=True时，只支持查询 单支股票 或 股票列表 的复权行情，而symbol='{symbol}'")

    # 数据库信息
    db_config = db_config or get_db_config(key='prices', hk=hk)
    area = area or decide_area(symbol)
    tb_name = tb_name or get_tb_name(symbol=symbol, area=area, db_name=db_name, key='px')

    if day:
        start = end = start
    parse_dates = ['date1']
    rename_col = {'date1': 'datetime'}

    # 所有股票
    if not symbol:
        sql_list = _sql_px_orig_all(tb_template=tb_name, fields=fields,
                                    day=day, start=start, end=end)
        return pd.concat(get_mysql_p(item_list=sql_list, con=db_config,
                                     parse_dates=parse_dates, rename_col=rename_col,
                                     progress=progress,
                                     nbr_thread=nbr_thread, chunksize=chunksize,
                                     desc=f'Getting original data of all symbols with {len(sql_list)} queries',
                                     **kwargs),
                         ignore_index=True)

    # 单个股票
    if isinstance(symbol, str):
        if incl_prev:
            new_start = get_symbol_prev_day(symbol=symbol, day=start,
                                            db_config=db_config, db_name=db_name, tb_name=tb_name,
                                            hk=hk, area=area, **kwargs)
            if progress and new_start:
                print(f'incl_prev implies the start {start} -> {new_start}')
            start = new_start or start

        sql_list = _sql_px_orig_single(symbol=symbol, tb_template=tb_name, fields=fields,
                                       day=day, start=start, end=end)
        return pd.concat(get_mysql_p(item_list=sql_list, con=db_config,
                                     parse_dates=parse_dates, rename_col=rename_col,
                                     progress=progress,
                                     nbr_thread=nbr_thread, chunksize=chunksize,
                                     desc=f'Getting original data of {symbol} with {len(sql_list)} queries',
                                     **kwargs),
                         ignore_index=True)

    # 股票列表
    if incl_prev:
        new_start_list = get_symbol_prev_day(symbol=symbol, day=start,
                                             db_config=db_config, db_name=db_name, tb_name=tb_name,
                                             hk=hk, area=area, **kwargs)
        sql_list_list = [_sql_px_orig_single(symbol=s, tb_template=tb_name, fields=fields, day=day,
                                             start=nst or start, end=end)
                         for s, nst in zip(symbol, new_start_list)]
    else:
        sql_list_list = _sql_px_orig_multiple(symbol_list=symbol, tb_template=tb_name, fields=fields,
                                              day=day, start=start, end=end)
    sql_list = list(chain.from_iterable(sql_list_list))
    break_index = [None, *cumsum([len(sql_list) for sql_list in sql_list_list])]  # 供合并用
    # dfq: dataframe of query
    dfq_list = get_mysql_p(item_list=sql_list,
                           con=db_config,
                           parse_dates=parse_dates, rename_col=rename_col,
                           progress=progress,
                           nbr_thread=nbr_thread, chunksize=chunksize,
                           desc=f'Getting original data of {len(symbol)} symbols with {len(sql_list)} queries',
                           **kwargs)
    return [pd.concat(dfq_list[i:j], ignore_index=True)
            for i, j in zip(break_index[:-1], break_index[1:])]

[docs]def get_adj(symbol: Union[str, list, tuple], start: str = None, end: str = None,
            db_config: dict = None, tb_name: str = None,
            hk: bool = False, area: str = None, progress: bool = False,
            full: bool = True,
            **kwargs):
    """
    获取复权数据，包括除息除权日、分红、系数等。

    :param symbol: 如'000001 CH Equity'；空('' 或 [] 或 () ) 代表所有股票；支持股票列表或股票元组。
    :param start: 起始日期（含），格式如'2019-01-01'；''代表数据库中最早的日期。
    :param end: 终止日期（含），格式如'2019-09-01'；''代表数据库中最晚的日期。
    :param db_config: 指定数据库连接，None则自动分配。
    :param tb_name: 指定的表名，格式为{db_name}.{tb_name}，None则自动分配。
    :param hk: 在香港使用需设置为True。
    :param area: in ['CH', 'US', ...]。指定symbol区域，不指定则根据symbol（或symbol list的第一个）判定。为空时，自动设置为'CH'。
    :param progress: 显示下载进度条。
    :param full: 是否下载所有列。
    :param kwargs:

    :return: a list/tuple of symbols -> DataFrame list; otherwise, DataFrame
    """
    try:
        if symbol.endswith('Index'):
            return pd.DataFrame()
    except AttributeError:
        pass

    # shortcut
    if not symbol:
        symbol = ''
    if len(symbol) > 50:  # 超过50个采用全部下载再过滤的方式
        df_all = get_adj(symbol='', start=start, end=end, db_config=db_config, tb_name=tb_name, hk=hk, area=area,
                         progress=progress, **kwargs)
        df_selected = df_all[df_all.symbol.isin(symbol)]
        df_dict = dict(list(df_selected.groupby(by='symbol')))
        return [df_dict.get(s) for s in symbol]

    area = area or decide_area(symbol)
    area = area.lower()
    tb_name = tb_name or get_tb_name(symbol=symbol, area=area, key='dvd')
    desc = 'Getting adjustments info'
    if area == 'ch':
        db_config = db_config or get_db_config(key='ivt', hk=hk)
        cols = [f'F00{i}0' for i in range(1, 9)]
        fields = ['SEC_CD', *cols]
        parse_dates = ['F0010']
        rename_col = {
                'SEC_CD': 'symbol',
                'F0010':  'ex_date',
                'F0020':  'adj_cap',
                'F0030':  'dvd_cash',
                'F0060':  'adj',
                'F0080':  'adj_cap_r'
        }
        df = get_fields(db_config=db_config, tb_name=tb_name, ivt=True,
                        fields=fields, symbol=s2code_list(symbol), symbol_col='SEC_CD',
                        start=start, end=end, dt_col='F0010',
                        # order_by='F0010 ASC',
                        parse_dates=parse_dates, rename_col=rename_col,
                        progress=progress,
                        desc=desc,
                        **kwargs
                        )
    else:  # overseas
        db_config = db_config or get_db_config(key='dvd_ovs', hk=hk)
        tb_name = tb_name or get_tb_name(key='dvd')
        if full:
            fields = '*'
        else:
            fields = ['symbol', 'ex_date', 'dvd_type', 'amount_all_fixed', 'adj']
        parse_dates = ['ex_date']
        rename_col = {
                'ex_date':                   'ex_date',  # 否则会被自动重命名为datetime
                'dividend_amount_all_fixed': 'amount'
        }
        df = get_fields(db_config=db_config, tb_name=tb_name, ivt=False,
                        fields=fields, symbol=symbol, start=start, end=end, dt_col='ex_date',
                        # order_by='ex_date ASC',
                        more='AND is_vld=1',  # 复权表要求
                        parse_dates=parse_dates, rename_col=rename_col, progress=progress,
                        desc=desc,
                        **kwargs)

        # 处理capital change相关
        df_list = df if isinstance(df, list) else [df]

        df_ae_list = []  # ae: after effects, add a column: adj_cap_r
        for df_raw in df_list:
            cap_change_flag = df_raw['dvd_type'].isin(cap_types + split_types)
            cap_change_flag = cap_change_flag | df_raw['note'].str.endswith('split')
            df_ae = df_raw.copy()
            df_ae['adj_cap_r'] = 1
            df_ae.loc[cap_change_flag, 'adj_cap_r'] = 1 / df_ae.loc[cap_change_flag, 'adj']
            df_ae = df_ae[df_ae.dvd_type.isin(valid_types)].reset_index(drop=True)

            df_ae_list.append(df_ae)

        if len(df_ae_list) == 1:
            df = df_ae_list[0]
        else:
            df = df_ae_list

    return df

    :param area: in ['CH', 'US', ...]。指定symbol区域，不指定则根据symbol（或symbol list的第一个）判定。为空时，自动设置为'CH'。
    :param progress: 显示下载进度条。
    :param kwargs:

    :return: one symbol -> one DataFrame; otherwise -> DataFrame list
    """
    if isinstance(symbol, (list, tuple)) and len(symbol) == 1:
        symbol = symbol[0]

    # 确定数据库信息
    px_db_config = px_db_config or get_db_config(key='prices', hk=hk)
    area = area or decide_area(symbol)
    px_tb_name = px_tb_name or get_tb_name(symbol=symbol, area=area, db_name=px_db_name, key='px')

    # 原始行情
    data_px = get_px_orig(symbol=symbol, fields=fields,
                          day=day, start=start, end=end, incl_prev=incl_prev,
                          db_config=px_db_config, db_name=px_db_name, tb_name=px_tb_name,
                          hk=hk, area=area, progress=progress,
                          **kwargs)

    try:
        if data_px.empty:
            return data_px
    except AttributeError:  # 此时data_px为列表
        if all(df.empty for df in data_px):
            return data_px

    # 复权数据
    if day:
        start_df_adj = day
        end_df_adj = ''  # 指定day时，adj_all强制为True
    else:
        start_df_adj = start
        end_df_adj = '' if adj_all else end
    if incl_prev:  # 此时下载df_adj的起点为最早的时间点
        if isinstance(data_px, list):
            start_df_adj = min(df['datetime'].iat[0] for df in data_px if not df.empty).date().isoformat()
        else:
            start_df_adj = data_px['datetime'].iat[0].date().isoformat()
    data_adj = get_adj(symbol=symbol, start=start_df_adj, end=end_df_adj,
                       db_config=adj_db_config, tb_name=adj_tb_name,
                       area=area, hk=hk, progress=progress, **kwargs)

    # NOTE: 如果symbol为list
    # data_px一定为list
    # adj_px在list中只有一个股票的时候，会是一个dataframe

    # 复权行情
    if not symbol:  # 所有股票, dfs_list
        df_orig_dict = dict(list(data_px.groupby('symbol')))
        df_adj_dict = dict(list(data_adj.groupby('symbol')))
        item_list = [(df_orig, df_adj_dict.get(s)) for s, df_orig in df_orig_dict.items()]
        df_px_adj_list = adj_px_by2df_p(item_list=item_list, progress=progress, adj_type=adj_type)
        return df_px_adj_list

    if isinstance(symbol, str):  # 单个股票, dfs
        return adj_px_by2df(df_df=(data_px, data_adj), adj_type=adj_type)

    # 股票列表, dfs_list
    return adj_px_by2df_p(item_list=list(zip(data_px, data_adj)), progress=progress, adj_type=adj_type)

    * 返回单个df、df列表或df字典。

    :param start: 起始日期（含），格式如'2019-01-01'；空('' 或 None) 则自动设置为1992-01-01。

    :param trim: 是否强制对齐到同一个交易日历。不强制时，指定的start之后上市的股票只对齐到其上市之后的交易日历。

    :param area: in ['CH', 'US', ...]。指定symbol区域，不指定则根据symbol（或symbol list的第一个）判定。为空时，自动设置为'CH'。
    :param fmt: 返回数据类型，支持DataFrame、DataFrame list、DataFrame dict. 默认为'df'。

    :param kwargs: 其他选项。

        symbol = symbol[0]

    # NOTE: 包括停牌时，需要下载start之前一个交易日的数据
    if suspend and (not symbol):
        raise NotImplementedError(f"当包括停牌时，只支持查询 单支股票 或 股票列表 的行情，而symbol='{symbol}'")

    # 确定数据库信息
    px_db_config = px_db_config or get_db_config(key='prices', hk=hk)
    area = area or decide_area(symbol)
    px_tb_name = px_tb_name or get_tb_name(symbol=symbol, area=area, db_name=px_db_name, key='px')

    # 获取数据
    if day:
        start = end = day

    # 全是指数时不用复权
    if symbol:
        try:
            all_index = symbol.endswith('Index')
        except AttributeError:  # 此时symbol为list
            all_index = all(s.endswith('Index') for s in symbol)
    else:
        all_index = False

    # 行情
    if all_index or adj_type == 'orig' or (not adj_type):  # 不复权
        data_px = get_px_orig(symbol=symbol, fields=fields,
                              day=day, start=start, end=end,
                              incl_prev=bool(suspend),  # 处理停牌则需要下载start前一个交易日信息
                              db_config=px_db_config, db_name=px_db_name, tb_name=px_tb_name,
                              hk=hk, area=area, progress=progress,
                              **kwargs)
    else:  # 复权
        data_px = get_px_adj(symbol=symbol, start=start, end=end, day=day,
                             fields=fields,

                             px_db_config=px_db_config, px_db_name=px_db_name, px_tb_name=px_tb_name,
                             adj_all=adj_all, adj_type=adj_type,
                             adj_db_config=adj_db_config, adj_tb_name=adj_tb_name,
                             hk=hk, area=area, progress=progress,
                             **kwargs)

    # 处理停牌
    if suspend:
        if not isinstance(data_px, list):
            data_px = [df for _, df in data_px.groupby('symbol')]
        calendar = get_calendar(area=area, start=start, end=end, hk=hk, **kwargs)
        data_px = align_date_p(item_list=data_px, calendar=calendar,
                               method=suspend, trim=trim, rt_start=start,

        if len(data_px) == 1:
            data_px = data_px[0]

    # 返回数据

    if fmt == 'df':
        if isinstance(data_px, list):

    if fmt == 'list':
        if not symbol:  # all symbols
            df_list = [df.set_index(index_col) if index_col else df
                       for _, df in data_px.groupby('symbol')]
        elif isinstance(symbol, str):  # one symbol

    if fmt == 'dict':

        elif isinstance(symbol, str):  # one symbol
