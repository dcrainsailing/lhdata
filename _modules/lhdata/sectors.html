@pack_dict
def _get_sector(symbol: str, system: str = None, level: int = 3, db_config: dict = None,
                df_static: pd.DataFrame = None) -> pd.DataFrame:
    """
    给定股票，查询所属行业。

    :param symbol: 如'000001 CH Equity'.
    :param system: 行业体系，in ['gics', 'exchange', 'investoday', 'swhy', 'citic']。None则自动分配： A股swhy， 外盘gics。
    :param level: 如果超出该体系的级别范围，返回该体系所有级别的对应信息
    :param db_config: 指定数据库连接，None则自动分配。
    :param df_static: 存储好的股票、代码表


    :return: DataFrame， 包含列：symbol、name、code_sector
    """
    if system is None:
        area = decide_area(symbol).lower()
        if area == 'ch':
            system = 'swhy'
        else:
            system = 'gics'
    else:
        system = system.lower()

    if system == 'swhy':
        idu_cl = sector_sys_map.get(system)
        symbol = symbol.split()[0]
        db_config = db_config or get_db_config(key='ivt')
        con = pymssql.connect(**db_config)

        code_str = f"concat(left({tb_symbol_indus}.{idu_cl}, {level * 2}), '{'0' * (6 - level * 2)}')" \
            if level in [1, 2] else f'{tb_symbol_indus}.{idu_cl}'

        sql_string = f"""SELECT {tb_symbol_info}.SEC_CD as 'symbol', 
                                {tb_symbol_info}.SEC_SNM as 'name', 
                                {code_str} as 'code', 
                                {tb_indus_info}.F0010 as 'sector'
                         From {tb_symbol_indus}
                            inner join {tb_symbol_info}
                                on {tb_symbol_indus}.ORG_ID = {tb_symbol_info}.ORG_ID
                            left join {tb_indus_info}
                                on {tb_indus_info}.IDU_CL = '{idu_cl}'
                                and {tb_indus_info}.INDUS_CL = {code_str}
                         WHERE {tb_symbol_indus}.IS_VLD = 1
                            AND {tb_indus_info}.IS_VLD = 1
                            AND {tb_symbol_info}.IS_VLD = 1
                            AND {tb_symbol_info}.SEC_CD = '{symbol}'"""

        try:
            df = get(sql_string=sql_string, con=con)
            df.symbol = df.symbol.apply(lambda x: f'{x} CH Equity')
        except Exception as e:
            raise Exception(f'Error getting {system} sector for {symbol} :\n{e}')

        if level not in [1, 2, 3]:  # all levels
            code = df.code.iat[0]
            sector_table = get_sector_table(system=system, db_config=db_config, transform=True)
            df2 = sector_table[sector_table.code == code].reset_index(drop=True)
            df = pd.concat([df[['symbol', 'name']], df2], axis=1)
        return df

    elif system == 'gics':
        if df_static is None:
            df_static = pd.read_csv(os.path.join(static_dir, 'all_gics.csv'))
            df_static.code = df_static.code.astype(str)
        df_symbol = df_static[df_static.symbol == symbol]
        if not df_symbol.empty:
            df_symbol = df_symbol.copy()
            if level not in [4, 3, 2, 1]:
                for lv in [4, 3, 2, 1]:
                    df_symbol[f'lv{lv}'] = df_symbol.code.apply(lambda x: x[:lv * 2])
            else:
                df_symbol.code = df_symbol.code.apply(lambda x: x[:level * 2])

        return df_symbol.reset_index(drop=True)

    else:
        pass


@pack_dict
def _get_sector_symbol(code: Union[str, int], system: str = None, db_config: dict = None,
                       df_static: pd.DataFrame = None, area: str = None) -> pd.DataFrame:
    """
    给定一个体系的行业代码，返回该行业涵盖的股票。

    :param code: 行业代码，如110101
    :param system: 行业体系，in ['gics', 'exchange', 'investoday', 'swhy', 'citic']。None则自动分配： A股swhy， 外盘gics。
    :param db_config: 指定数据库连接，None则自动分配。
    :param df_static: 存储好的股票、代码表
    :param area: like 'us'. 当system='gics'的时候有用, 或用来确定sys。None则不对区域做限制。

    :return: DataFrame
    """
    code = str(code)
    area = area or 'ch'
    if system is None:
        system = 'swhy' if area == 'ch' else 'gics'
    else:
        system = system.lower()
    system = system or 'swhy'

    if system == 'swhy':
        idu_cl = sector_sys_map.get(system)
        db_config = db_config or get_db_config('ivt')
        con = pymssql.connect(**db_config)

        if code.endswith('0000'):
            code_str = f"{tb_symbol_indus}.INDUS4_CL LIKE '{code[:2]}____'"
        elif code.endswith('00'):
            code_str = f"{tb_symbol_indus}.INDUS4_CL LIKE '{code[:4]}__'"
        else:
            code_str = f"{tb_symbol_indus}.INDUS4_CL = {code}"

        sql_string = f"""SELECT {tb_symbol_info}.SEC_CD as 'symbol', 
                                {tb_symbol_info}.SEC_SNM as 'name', 
                                {tb_symbol_indus}.{idu_cl} as 'code', 
                                {tb_indus_info}.F0010 as 'sector'
                         From {tb_symbol_indus}
                            inner join {tb_symbol_info}
                                on {tb_symbol_indus}.ORG_ID = {tb_symbol_info}.ORG_ID
                            left join {tb_indus_info}
                                on {tb_indus_info}.IDU_CL = '{idu_cl}'
                                and {tb_indus_info}.INDUS_CL = {tb_symbol_indus}.{idu_cl}
                         WHERE {tb_symbol_indus}.IS_VLD = 1
                            AND {tb_indus_info}.IS_VLD = 1
                            AND {tb_symbol_info}.IS_VLD = 1
                            AND {tb_symbol_info}.VAR_CL in ('a') and {tb_symbol_info}.MKT_CL in ('s','z') 
                            AND {code_str}"""
        try:
            df = get(sql_string=sql_string, con=con)
            df.symbol = df.symbol.apply(lambda x: f'{x} CH Equity')
        except Exception as e:
            raise Exception(f'Error getting info for sector {code}:\n{e}')

        return df
    elif system == 'gics':
        if df_static is None:
            df_static = pd.read_csv(os.path.join(static_dir, 'all_gics.csv'))
            df_static.code = df_static.code.astype(str)
        df_sector = df_static[df_static.code.str.startswith(code)]
        if area:
            df_sector = df_sector[df_sector.symbol.apply(decide_area) == area.upper()]
        return df_sector.reset_index(drop=True)
    else:
        pass

[docs]def get_sector_table(system: str = None,
                     max_lv: int = None,
                     transform: bool = True,
                     area: str = 'ch',
                     db_config: dict = None, tb_name: str = None,
                     static: bool = False) -> pd.DataFrame:
    """
    指定体系的行业分类代码表。由今日投资维护。

    :param system: 行业体系，in ['gics', 'exchange', 'investoday', 'swhy', 'citic']。None则自动分配： A股swhy， 外盘gics。
    :param max_lv: 最大到什么级别，None为不限制。
    :param transform: 查询到的表各个级别的code混在一起，将表转换为更易读的形式。
    :param area: in ['ch', 'us', 'jp', 'hk', 'ovs'].
    :param db_config: 指定数据库连接，None则自动分配。
    :param tb_name: 指定表名，None则自动分配。
    :param static: 是否使用已存储的副本。如果无法连接到服务器会自动使用。

    :return: DataFrame
    """
    if system is None:
        system = 'swhy' if area == 'ch' else 'gics'
    else:
        system = system.lower()

    df = None
    if not static:
        db_config = db_config or get_db_config('ivt')
        tb_name = tb_name or get_tb_name(symbol='', area=area, key='sector')

        fields = ["IDU_LEVEL", "INDUS_CL", "F0010", "F0030", "RMRK", "UPDT_TM"]

        idu_cl = sector_sys_map.get(system)
        more_str = f"AND IDU_CL = '{idu_cl}'"

        rename_col = {
                'IDU_LEVEL': 'level',
                'UPDT_TM':   'updatetime',
                'INDUS_CL':  'code',
                'F0010':     'sector',
                'F0030':     'sector_eng',
                'RMRK':      'system'}

        try:
            df = get_fields(db_config=db_config, tb_name=tb_name, ivt=True,
                            fields=fields, more=more_str, order_by='INDUS_CL',
                            parse_dates=['UPDT_TM'], rename_col=rename_col)
        except:
            print(f'Error getting sector table of {system} from database {tb_name}.')
            print(f'Try loading from stored version...')

    if df is None:
        fp = os.path.join(static_dir, f'tb_{system}.csv')
        df = pd.read_csv(fp, parse_dates=['updatetime'])
        df.code = df.code.astype(str)

    if max_lv is not None:
        df = df[df.level <= max_lv]

    if transform:
        df = _transform_sector_table(df, system=system)
    return df



def __code2high_lv(code: str, lv: int, system: str):
    if system == 'swhy':
        if code.endswith('0000'):
            cur_lv = 1
        elif code.endswith('00'):
            cur_lv = 2
        else:
            cur_lv = 3

        if cur_lv == 1 or cur_lv <= lv:
            return code

        if lv == 1:
            return f'{code[:2]}0000'
        if lv == 2:
            return f'{code[:4]}00'

    if system == 'gics':
        cur_lv = len(code) / 2
        if cur_lv == 1 or cur_lv <= lv:
            return code
        return code[:lv * 2]


def _transform_sector_table(df: pd.DataFrame, system: str, lv: int = None) -> pd.DataFrame:
    """
    整理下载好的行业分类表

    :param df: 下载的原始行业分类表
    :param system: 行业体系，in ['gics', 'exchange', 'investoday', 'swhy', 'citic']

    :return: DataFrame
    """
    if lv is None:
        lv = df.level.max()
    df_lv = df.loc[df['level'] == lv, ['code', 'sector', 'system']].rename(columns={'sector': f'lv{lv}'})
    for lv in range(lv - 1, 0, -1):
        df_lv_temp = df[df['level'] == lv].set_index('code')
        df_lv[f'lv{lv}'] = df_lv['code'].apply(lambda c: df_lv_temp.at[__code2high_lv(c, lv, system), 'sector'])
    df_lv = df_lv[['code', *[i for i in df_lv.columns if i.startswith('lv')], 'system']].reset_index(drop=True)
    return df_lv
