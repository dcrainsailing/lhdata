             db_config: dict = None, tb_name: str = None,
             hk: bool = False,
             **kwargs):
    """
    获取指数成分股。

    * 沪深300、中证500 由今日投资维护。
    * 创业板指、恒指、国企指数、道指、纳斯达克100、标普500、日经225 由Lighthorse维护。

    :param index: 指数名字，如'hs300'. 支持上述6个指数，可以使用中、英、别名。建议使用指数对应的ticker，如hs300对应SHSZ300 Index。
    :param day: 指定日期，会忽略start、end的输入。若day、start、end都为空，则设置为查询当日。注意此时当日数据可能还未准备好。
    :param start: 起始日期（含），格式如'2019-01-01'；空('' 或 None) 则自动设置为1992-01-01。
    :param end: 终止日期（含），格式如'2019-09-01'；空('' 或 None) 则自动设置为查询当日。
    :param db_config: 指定 成分表 数据库连接，None则自动分配。
    :param tb_name: 指定 成分表 的表名，格式为{db_name}.{tb_name}，None则自动分配。
    :param hk: 在香港使用需设置为True。
    :param kwargs: 其他get_fields支持传入的参数。

    :return: DataFrame
    """
    index_key = idx2idx_key(name=index)
    if all(i is None for i in (start, end, day)):
        day = pd.Timestamp.now().date().isoformat()

    if index_key in {'000300', '000905'}:
        db_config = db_config or get_db_config(key='ivt', hk=hk)
        tb_name = tb_name or '[dbo].[IDX_D_WEIGHT]'
        fields = ['PUB_DT', 'SEC_CD']
        rename_col = {'PUB_DT': 'datetime', 'SEC_CD': 'symbol'}
        df = get_fields(db_config=db_config, tb_name=tb_name, ivt=True, fields=fields,
                        symbol=index_key, symbol_col='IDX_CD',
                        start=start, end=end, day=day, dt_col='PUB_DT',
                        order_by='PUB_DT ASC', parse_dates=['PUB_DT'], rename_col=rename_col,
                        **kwargs)
    else:
        db_config = db_config or get_db_config(key='memb', hk=hk)
        if tb_name:
            tb_name_list = [tb_name]
        else:
            year_list = []
            if day:
                start = end = day
            year_related = [pd.Timestamp(x).year for x in (day, start, end)]
            if any(y < 2011 for y in year_related):
                year_list.append(1990)
            if any(y >= 2011 for y in year_related):
                year_list.append(2011)
            tb_name_list = ["quotation_hk.indxmember_hist_{y}".format(y=y) for y in year_list]

        fields = ['date', 'symbol']
        rename_col = {'date': 'datetime'}
        if day or (start == end):  # 单日按股票排序
            order_by = 'symbol ASC'
        else:  # 多日按日期排序
            order_by = 'date ASC'
        df_list = [get_fields(db_config=db_config, tb_name=tn, ivt=False, fields=fields,
                              symbol=index_key, symbol_col='indexi',
                              start=start, end=end, day=day, dt_col='date',
                              order_by=order_by, parse_dates=['date'], rename_col=rename_col,
                              **kwargs)
                   for tn in tb_name_list]
        df = pd.concat(df_list)
    df['idx_symbol'] = idx2symbol(idx2symbol(idx_key=index_key))
    df = df[['idx_symbol', 'datetime', 'symbol']]
    return df
