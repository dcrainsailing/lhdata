from .dbgetter import get_fields
from .dispatcher import get_db_config, get_tb_name, WF_FEATURES
from .util_data import decide_area, idx2symbol
from typing import Union
import pandas as pd


def _get_features(symbol: Union[str, list, tuple] = None, fmt: str = 'df',
                  **kwargs):
    results = get_fields(symbol=symbol, **kwargs)

    if fmt == 'df':
        return pd.concat(results) if isinstance(results, list) else results

    if fmt == 'list':
        return results if isinstance(results, list) else [results]

    if fmt == 'dict':
        if not symbol:  # 所有股票
            return {s: df for s, df in results.groupby('symbol')}
        if isinstance(symbol, str):
            return {symbol: results}
        # return {df.symbol.iat[0]: df
        #         for df in results if (df is not None) and (not df.empty)}
        return {s: df for s, df in zip(symbol, results)}



[docs]def get_alpha(symbol: Union[str, list, tuple] = '',
              index_ref: str = None,
              day: str = None, start: str = None, end: str = None,
              fields: Union[str, list, tuple] = '*', order_by: str = 'datetime',
              db_config: dict = None, tb_name: str = None,
              area: str = None, hk: bool = False, fmt: str = 'df',
              **kwargs):
    """
    获取Alpha Features。

    :param symbol: 如'000001 CH Equity'；空('' 或 [] 或 () ) 代表所有股票；支持股票列表或股票元组。
    :param index_ref: A股：['hs300', 'zz500', 'gemb']; US: ['indu', 'spx', 'ndx']; HK: ['hsi', 'hscei']; JP: ['nky'].
                    None则自动选择默认，A股默认'hs300', 美股默认'spx', 港股默认'hsi'。
    :param start: 起始日期（含），格式如'2019-01-01'；空('' 或 None) 则自动设置为1992-01-01。
    :param end: 终止日期（含），格式如'2019-09-01'；空('' 或 None) 则自动设置为查询当日。
    :param day: 指定日期，等效于设置了start=end=day, 会忽略start、end的输入。
    :param fields: 指定'datetime'、'symbol'以外的字段，默认所有字段。
    :param order_by: 默认按日期排序
    :param db_config: 指定 特征 数据库连接，None则自动分配。
    :param tb_name: 指定 特征 表名，格式为{db_name}.{tb_name}，None则自动分配。
    :param area: in ['CH', 'US', ...]。指定symbol区域，不指定则根据symbol（或symbol list的第一个）判定。为空时，自动设置为'CH'。
    :param hk: 在香港使用需设置为True。
    :param fmt: 返回数据类型，支持['df', 'list', 'dict']，即DataFrame、DataFrame list、DataFrame dict. 默认为'df'。
    :param kwargs: 其他lhdata.dbgetter.get_fields支持传入的参数。

    :return: DataFrame 或 DataFrame list 或 DataFrame dict
    """
    if not area:
        if symbol:
            area = decide_area(symbol)
        else:
            area = decide_area(idx2symbol(index_ref))
    area = area.lower()

    if not index_ref:
        if area == 'ch':
            index_ref = 'hs300'
        elif area == 'us':
            index_ref = 'spx'
        elif area == 'hk':
            index_ref = 'hsi'
        elif area == 'jp':
            index_ref = 'nky'
        else:
            index_ref = 'hs300'

    db_config = db_config or get_db_config(key='alpha', hk=hk)
    tb_name = tb_name or get_tb_name(symbol=symbol, area=area, key='alpha').format(index_ref=index_ref,
                                                                                   area=area)

    return _get_features(symbol=symbol, fmt=fmt,
                         db_config=db_config, tb_name=tb_name, ivt=False,
                         fields=fields, must_have=['datetime', 'symbol'], order_by=order_by,
                         day=day, start=start, end=end, parse_dates=['datetime'], **kwargs)

           live: bool = False,
           db_config: dict = None, tb_name: str = None,
           area: str = None, hk: bool = False, fmt: str = 'df',
           **kwargs):
    """
    获取Wave Features。

    :param symbol: 如'000001 CH Equity'；空('' 或 [] 或 () ) 代表所有股票；支持股票列表或股票元组。
    :param start: 起始日期（含），格式如'2019-01-01'；空('' 或 None) 则自动设置为1992-01-01。
    :param end: 终止日期（含），格式如'2019-09-01'；空('' 或 None) 则自动设置为查询当日。
    :param day: 指定日期，等效于设置了start=end=day, 会忽略start、end的输入。
    :param fields: 指定'datetime'、'symbol'以外的字段，默认为WF0010~WF0060。
    :param incl_week: 是否自动添加week后缀的feature。默认不添加。
    :param incl_val: 是否自动添加feature的原始值。默认不添加。
    :param live: True -> 盘中值; False -> 盘后值。默认为盘后值。
    :param db_config: 指定 特征 数据库连接，None则自动分配。
    :param tb_name: 指定 特征 表名，格式为{db_name}.{tb_name}，None则自动分配。
    :param area: in ['CH', 'US', ...]。指定symbol区域，不指定则根据symbol（或symbol list的第一个）判定。为空时，自动设置为'CH'。
    :param hk: 在香港使用需设置为True。
    :param fmt: 返回数据类型，支持['df', 'list', 'dict']，即DataFrame、DataFrame list、DataFrame dict. 默认为'df'。
    :param kwargs: 其他lhdata.dbgetter.get_fields支持传入的参数。

    :return: DataFrame 或 DataFrame list 或 DataFrame dict
    """
    # db
    area = area or decide_area(symbol)
    db_config = db_config or get_db_config(key='wf', hk=hk)
    tb_name = tb_name or get_tb_name(symbol=symbol, area=area, key='wf', live=live)

    cols = ['WF0010', 'WF0011', 'WF0020', 'WF0030', 'WF0031', 'WF0032', 'WF0040', 'WF0041', 'WF0050', 'WF0060',
            'WF0010_val', 'WF0011_val', 'WF0020_val', 'WF0030_val',
            'WF0040_val', 'WF0041_val', 'WF0050_val', 'WF0060_val',
            'smoothed', 'atr', 'y_linear', 'std', 'w_up', 'w_down',

            'WF0010_week', 'WF0011_week', 'WF0020_week', 'WF0030_week', 'WF0031_week', 'WF0032_week',
            'WF0040_week', 'WF0041_week', 'WF0050_week', 'WF0060_week',
            'WF0010_val_week', 'WF0011_val_week', 'WF0020_val_week', 'WF0030_val_week',
            'WF0040_val_week', 'WF0041_val_week', 'WF0050_val_week', 'WF0060_val_week',
            'y_linear_week', 'std_week', 'w_up_week', 'w_down_week',

            'UPTIME']
    # fields
    if fields == '*':
        fields = WF_FEATURES

    fields = [fields] if isinstance(fields, str) else list(fields)

    if incl_val:
        fields += [f'{f}_val' for f in fields]
    if incl_week:
        fields += [f'{f}_week' for f in fields]

    fields = [f for f in fields if f in cols]
    must_have = ['DATE', 'SYMBOL']
    if live:
        must_have.append('UPTIME')

    return _get_features(symbol=symbol, fmt=fmt,
                         db_config=db_config, tb_name=tb_name, ivt=False,
                         fields=fields, must_have=must_have,
                         day=day, start=start, end=end, dt_col='DATE', parse_dates=['DATE'],
                         symbol_col='SYMBOL', rename_col={'UPTIME': 'updatetime'},
                         **kwargs)
