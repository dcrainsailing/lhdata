[docs]def get_calendar(area: str = "CH", start: str = None, end: str = None,
                 db_config: dict = None, db_name: str = None, tb_name: str = None,
                 holiday_only: bool = False, fmt: str = 'index',
                 hk: bool = False, **kwargs):
    """
    指定市场，获得该市场的交易日历。

    :param area: in ['CH', 'US', 'HK', 'JP']. 默认为'CH'. 支持大小写混用。
    :param start: 起始日期（含），格式如‘2019-01-01’；空('' 或 None) 则自动设置为1992-01-01。
    :param end: 终止日期（含），格式如‘2019-09-01’；空('' 或 None) 则自动设置为查询当日。
    :param db_config: 指定 交易日历 数据库连接，None则自动分配。
    :param db_name: 指定 交易日历 数据库名，None则自动分配。
    :param tb_name: 指定 交易日历 表名，None则自动分配。
    :param holiday_only: 只过滤假期，不过滤临时停牌则选择True。默认为False。
    :param fmt: 返回格式。'index' -> DatetimeIndex; 'str' -> string list; otherwise, Timestamp list.
    :param hk: 在香港使用需设置为True。
    :param kwargs: 其他get_fields支持传入的参数。

    :return: DatetimeIndex 或 string list 或 Timestamp list
    """
    df_holidays = get_holidays(area=area, start=start, end=end,
                               db_config=db_config, db_name=db_name, tb_name=tb_name,
                               holiday_only=holiday_only, hk=hk, **kwargs)

    # 过滤休市日
    start = start or DEFAULT_START
    end = end or pd.Timestamp.now().date().isoformat()
    business_days = pd.date_range(start=start, end=end, freq='B')
    trading_days = sorted(list(set(business_days) - set(df_holidays.closed)))

    # 格式
    if 'index' in fmt:
        return pd.DatetimeIndex(trading_days, name='datetime')
    if 'str' in fmt:
        return [d.date().isoformat() for d in trading_days]
    else:
        return trading_days




[docs]def get_holidays(area: str = 'CH', start: str = None, end: str = None,
                 db_config: dict = None, db_name: str = None, tb_name: str = None,
                 holiday_only: bool = False,
                 hk: bool = False, **kwargs) -> pd.DataFrame:
    """
    获取某市场的除周六日外的非交易日。

    :param area: in ['CH', 'US', 'HK', 'JP']. 默认为'CH'. 支持大小写混用。
    :param start: 起始日期（含），格式如‘2019-01-01’；空('' 或 None) 则自动设置为1992-01-01。
    :param end: 终止日期（含），格式如‘2019-09-01’；空('' 或 None) 则自动设置为查询当日。
    :param db_config: 指定 交易日历 数据库连接，None则自动分配。
    :param db_name: 指定 交易日历 数据库名，None则自动分配。
    :param tb_name: 指定 交易日历 表名，None则自动分配。
    :param holiday_only: 只返回假期，不返回临时停牌则选择True。默认为False。
    :param hk: 在香港使用需设置为True。
    :param kwargs: 其他get_fields支持传入的参数。

    :return: DataFrame
    """
    # 确定休市日表地址
    db_config = db_config or get_db_config(key='calendar', hk=hk)
    if not db_name:
        area = area.lower()
        if area == 'ch':
            db_name = 'quotation_sz'
        else:
            db_name = 'quotation_hk'
    tb_name = tb_name or f'{db_name}.tb_holiday_{area}'

    start = start or DEFAULT_START
    end = end or pd.Timestamp.now().date().isoformat()
    # 休市日表
    fields = ['date1', 'nonsettle_type']
    df_holidays = get_fields(db_config=db_config, tb_name=tb_name, fields=fields,
                             start=start, end=end, dt_col='date1', order_by='date1 ASC', parse_dates=['date1'],
                             rename_col={'date1': 'closed', 'nonsettle_type': 'type_'},
                             **kwargs)
    if holiday_only:
        df_holidays = df_holidays[df_holidays.type_ == 'holiday']
    return df_holidays




[docs]def get_prev_day(day: str, area: str = 'ch', space: int = 30,
                 db_config: dict = None, db_name: str = None, tb_name: str = None,
                 hk: bool = False, **kwargs):
    """
    指定市场，获得某日之前的一个交易日。通过查询交易日历实现。

    :param day: 格式如‘2019-01-01’。
    :param area: in ['CH', 'US', 'HK', 'JP']. 默认为'CH'. 支持大小写混用。
    :param space: 搜索空间。默认为day之前30个自然日。
    :param db_config: 指定 交易日历 数据库连接，None则自动分配。
    :param db_name: 指定 交易日历 数据库名，None则自动分配。
    :param tb_name: 指定 交易日历 表名，None则自动分配。
    :param hk: 在香港使用需设置为True。
    :param kwargs: 其他get_calendar支持传入的参数。

    :return: date str or None
    """
    start = (pd.Timestamp(day) - pd.Timedelta(days=space)).date().isoformat()
    calendar = get_calendar(area=area, start=start, end=day,
                            db_config=db_config, db_name=db_name, tb_name=tb_name,
                            fmt='index', hk=hk, **kwargs)
    calendar = calendar[calendar < day]
    try:
        return calendar[-1].date().isoformat()
    except IndexError:
        return None




[docs]def get_symbol_prev_day(symbol: Union[str, list, tuple], day: str,
                        db_config: dict = None, db_name: str = None, tb_name: str = None,
                        hk: bool = False, area: str = None,
                        **kwargs):
    """
    指定股票和日期，获得前一个正常交易日。通过查询行情表实现。

    :param symbol: 格式如'AAPL US Equity'。支持股票列表或股票元组。
    :param day: 格式如‘2019-01-01’。
    :param db_config: 指定 交易日历 数据库连接，None则自动分配。
    :param db_name: 指定 交易日历 数据库名，None则自动分配。
    :param tb_name: 指定 交易日历 表名，None则自动分配。
    :param hk: 在香港使用需设置为True。
    :param area: in ['CH', 'US', 'HK', 'JP']. 默认为'CH'. 支持大小写混用。
    :param kwargs: 其他lhdata.dbgetter.get_prev_支持传入的参数。

    :return: a list/tuple of symbols -> date str list; otherwise, date str
    """
    if not symbol:
        raise NotImplementedError(f'只支持查询 单支股票，或 股票列表 的前一个正常交易日，而symbol={symbol}')

    symbol = [symbol] if isinstance(symbol, str) else list(symbol)
    df_list = get_prev__ps(item_list=symbol, day=day, fields=['date1'],
                           db_config=db_config, db_name=db_name, tb_name=tb_name,
                           hk=hk, area=area, **kwargs)
    day_list = [df.iat[0, 0].date().isoformat() if df is not None else None
                for df in df_list]

    if len(day_list) == 1:
        return day_list[0]
    else:
        return day_list
