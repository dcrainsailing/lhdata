[docs]def get_ipo_day(symbol: Union[str, list, tuple],
                db_config: dict = None, tb_name: str = None,
                area: str = None, hk: bool = HK,
                **kwargs) -> pd.DataFrame:
    """
    获取ipo日期。注意外盘股票有可能上市日期为空。

    :param symbol: 如'000001 CH Equity'；只支持单只股票，或股票列表(元组)。
    :param db_config: 指定行情数据库连接，None则自动分配。
    :param tb_name: 指定行情表名，格式为{db_name}.{tb_name}，None则自动分配。
    :param area: in ['CH', 'US', ...]。指定symbol区域，不指定则根据symbol（或symbol list的第一个）判定。为空时，自动设置为'CH'。
    :param hk: 在香港使用需设置为True。
    :param kwargs: 其他dbgetter.get_fields支持传入的参数

    :return: DataFrame
    """
    try:
        if symbol.endswith('Index'):
            lookup_tb = {
                    'hs300': '2005-04-08',
                    'zz500': '2005-01-01',
                    'gemb': '2010-06-01'
            }
            df = pd.DataFrame(data=[[symbol, pd.Timestamp(lookup_tb.get(idx2name(symbol)))]],
                              columns=['symbol', 'ipo'])
            return df
    except:
        pass

    area = area or decide_area(symbol)
    area = area.lower()

    tb_name = tb_name or get_tb_name(key='basics', area=area)
    if area == 'ch':
        symbol_list = s2code_list(symbol)
        db_config = db_config or get_db_config(key='ivt')
        df = get_fields(db_config=db_config, tb_name=tb_name, ivt=True,
                        fields=['SEC_CD', 'MKT_DT'],
                        symbol=symbol_list, symbol_col='SEC_CD',
                        parse_dates=['MKT_DT'], rename_col={'MKT_DT': 'ipo'},
                        **kwargs)
    else:
        db_config = get_db_config(key='memb', hk=hk)
        df = get_fields(db_config=db_config, tb_name=tb_name, ivt=False,
                        fields=['symbol', 'ipo_date'],
                        symbol=symbol, parse_dates=['ipo_date'], rename_col={'ipo_date': 'ipo'},
                        **kwargs)

    return df

    tb_name = tb_name or get_tb_name(key='basics', area=area)

    df = get_fields(db_config=db_config, tb_name=tb_name, ivt=True,

                    **kwargs)
    return df



[docs]def get_symbols(area: str = 'ch', active: bool = True,
                type_: str = 'A',  board: str = None,  # for China
                order_by_symbol: bool = True,
                start: str = None, end: str = None, day: str = None,  # ipo constraints
                db_config: dict = None, tb_name: str = None,
                hk: bool = HK, **kwargs) -> pd.DataFrame:
    """
    获取某个市场所有的（外盘为订阅的）股票信息。

    :param area: in ['ch', 'us', 'hk', 'jp']。
    :param active: 是否正常交易。A股判断标准是退市日期为空，外盘通过market_status确认。
    :param type_: 品种代码, in ['A', 'B'], None则包括A和B股。只对A股有效。
    :param board: 板块代码, in ['MAIN'(主板), 'SME'(中小板), 'GEM'(创业板), 'STAR'(科创板)], None则所有版块。只对A股有效。
    :param order_by_symbol: True -> 按股票代码排序, False -> 按上市日期排序。
    :param start: 只返回该日期及之后上市的股票。None则不做限制。
    :param end: 只返回该日期及之前上市的股票。None则不做限制。
    :param day: 只返回该日上市的股票。None则不做限制。
    :param db_config: 指定行情数据库连接，None则自动分配。
    :param tb_name: 指定行情表名，格式为{db_name}.{tb_name}，None则自动分配。
    :param hk: 在香港使用需设置为True。
    :param kwargs: 其他dbgetter.get_fields支持传入的参数

    :return: DataFrame
    """
    area = area.lower()
    if area == 'ch':
        db_config = db_config or get_db_config(key='ivt', hk=hk)
        tb_name = tb_name or get_tb_name(symbol='', area=area, key='basics')
        fields = ['SEC_CD', 'SEC_NM', 'MKT_CL', 'MKT_DT', 'BOARD_CL']
        parse_dates = ['MKT_DT']
        rename_col = {
                'SEC_CD': 'symbol',
                'SEC_NM': 'name',
                'MKT_CL': 'exchange',
                'MKT_DT': 'ipo',
                'BOARD_CL': 'board'
        }

        if order_by_symbol:
            order_by_str = 'SEC_CD'
        else:
            order_by_str = 'MKT_DT'

        more_str = 'AND MKT_DT IS NOT NULL'
        if type_:
            more_str += f"\nAND VAR_CL = '{type_}'"
        if board:
            more_str += f"\nAND BOARD_CL = '{board}'"
        if active:
            more_str += f"\nAND DEL_DT IS NULL"
        else:  # 需要包含delist date
            fields.append('DEL_DT')
            parse_dates.append('DEL_DT')
            rename_col['DEL_DT'] = 'delisted'

        df = get_fields(db_config=db_config, tb_name=tb_name, ivt=True,
                        fields=fields,
                        start=start, end=end, day=day, dt_col='MKT_DT',
                        more=more_str, order_by=order_by_str,
                        parse_dates=parse_dates, rename_col=rename_col,
                        **kwargs)
        return df
    else:
        db_config = db_config or get_db_config(key='prices', hk=hk)
        tb_name = tb_name or get_tb_name(symbol='', area=area, key='basics')
        fields = ['symbol', 'name', 'exchangeid', 'ipo_date', 'market_status']
        parse_dates = ['ipo_date']
        rename_col = {
                'exchangeid': 'exchange',
                'ipo_date': 'ipo',
                'market_status': 'status'
        }

        if order_by_symbol:
            order_by_str = 'symbol'
        else:
            order_by_str = 'ipo_date'

        more_str = f"AND area = '{area}'\nAND sec_type = 'EQUITY'"
        if active:
            more_str += f"\nAND market_status = 'ACTV'"

        df = get_fields(db_config=db_config, tb_name=tb_name, ivt=False,
                        fields=fields,
                        start=start, end=end, day=day, dt_col='ipo_date',
                        more=more_str, order_by=order_by_str,
                        parse_dates=parse_dates, rename_col=rename_col,
                        **kwargs
                        )
        return df
